def filtered_star_assigner (row):
   if row['st_teff'] >= 5500 and row['st_teff'] <= 6300 and row['st_lum'] >= -0.5 and row['st_lum'] <= 0.5:
      return 'red'
   return 'blue'

def filtered_star_assigner_value (row, column):
    if row[column] == 'red':
        return 1
    elif row[column] == 'blue':
        return 0
    elif row[column] == 'green':
        return 2
    
dataset['filtered_color'] = dataset.apply (lambda row: filtered_star_assigner(row), axis=1)
dataset['filtered_color_value'] = dataset.apply (lambda row: filtered_star_assigner_value(row, 'filtered_color'), axis=1)
dataset = dataset.sort_values('filtered_color_value')

display_dataset = dataset[(dataset['filtered_color'] == 'red')].copy()
display_dataset.rename(columns = {'st_teff':'Temperature', 'st_lum':'Luminocity'}, inplace = True)
display_dataset.drop_duplicates(subset ="hostname",keep = False, inplace = True)
